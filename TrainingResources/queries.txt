USE tuto;

4. Create tables
CREATE TABLE product (
    idProduct INT AUTO_INCREMENT,
    name VARCHAR(20),
    price FLOAT,
    PRIMARY KEY (idProduct));
    
CREATE TABLE sale ( 
    idSale INT AUTO_INCREMENT,
    quantity INT,
    product INT,
    PRIMARY KEY (idsale),
    FOREIGN KEY (product) REFERENCES product(idProduct) ON DELETE CASCADE );
